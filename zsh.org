Set Java to 11 by default

#+BEGIN_SRC sh
  
  eval "$(/opt/homebrew/bin/brew shellenv)"

  export JAVA_11_HOME=$(/usr/libexec/java_home -v11)

  alias java11='export JAVA_HOME=$JAVA_11_HOME'

  # default to Java 11
  java11

#+END_SRC
